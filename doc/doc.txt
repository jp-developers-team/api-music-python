            $ pip install requirements

    # Heroku config "Postgres"

        # All addons

            https://elements.heroku.com/addons

        # Heroku Postgres

            https://elements.heroku.com/addons/heroku-postgresql

        # primeiro criar o projeto no heroku

            $ heroku login

            $ heroku create apimusicpython

        # Rodar o comando para criar a integração com o banco de dados "Postgres"

            $ heroku addons:create heroku-postgresql:hobby-dev

        # Por último fazer o "PUSH HEROKU" do REPOSITÓRIO GITHUB para o REPOSITÓRIO HEROKU

            $ git push heroku main

        if _name_ == '__main__':

        web: gunicorn app:app
